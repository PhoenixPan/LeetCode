public int firstBadVersion(int n) {
    if (n == 0 || !isBadVersion(n)) return 0;

    int start = 1;
    int end = n;
    while(start < end) {
        int mid = start + (end - start) / 2;
        if (isBadVersion(mid))
            end = mid;
        else 
            start = mid + 1;
    }
    // return isBadVersion(end)? end : 0;  
    return end; // or start
}
