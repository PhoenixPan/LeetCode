```java
public int numDecodings(String s) {
    int len = s.length();
    if (len == 0) return len;

    int[] dp = new int[len + 1];
    dp[0] = 1;
    dp[1] = s.charAt(0) == '0'? 0 : 1;

    for(int i = 2; i <= len; i++) {
        int one = Integer.parseInt(s.substring(i-1, i));
        int two = Integer.parseInt(s.substring(i-2, i));
        if (one > 0 && one < 10)
            dp[i] = dp[i - 1];
        if (two > 9 && two < 27)
            dp[i] += dp[i - 2];
    }
    return dp[len];
}
```
