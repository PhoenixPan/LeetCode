class Solution {
    public ListNode reverseList(ListNode head) {
        ListNode cur = head;
        ListNode newHead = null;
        while (cur != null) {
            ListNode temp = cur.next;
            if (newHead != null) {
                cur.next = newHead;
                newHead = cur;
            } else {
                newHead = cur;
                newHead.next = null;
            }
            cur = temp;
        }
        return newHead;
    }
}
