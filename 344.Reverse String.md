public String reverseString(String s) {
    // if (s.length() < 2) return s; // no impact on performance  
    char[] A = s.toCharArray();
    int l = 0;
    int r = A.length - 1;
    while (l < r) {
        char temp = A[l];
        A[l] = A[r];
        A[r] = temp;
        l++;
        r--;
    }
    return String.valueOf(A);
}
